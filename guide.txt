SELECT 
  locations.city, 
  locations.uf, 
  locations.uf_abbr, 
  locations.country, 
  locations.country_abbr, 
  courses.name, 
  universities.name, 
  universities.abbr
FROM 
  public.universities, 
  public.courses, 
  public.locations
WHERE 
  universities.location_id = locations.id AND
  courses.university_id = universities.id
ORDER BY
  universities.name ASC;
